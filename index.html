🎯 Number Guessing Game

Guess a number between 1 and 10

Score: 0

from js import document import random # pick a random number secret = random.randint(1, 10) score = 0 def check_guess(event): global secret, score guess = int(document.getElementById("guess").value) output = document.getElementById("output") score_display = document.getElementById("score") if guess == secret: score += 1 output.innerHTML = f"✅ Correct! The number was {secret}. A new number has been chosen." secret = random.randint(1, 10) # reset the number else: output.innerHTML = "❌ Wrong guess, try again!" score_display.innerHTML = str(score) # hook the button document.getElementById("submit").addEventListener("click", check_guess)
